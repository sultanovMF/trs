\end{equation}

  (2 - p(x_i) h) \cdot u_{i-1} + (2q(x_i)h^2 - 4) \cdot u_i + (2 + p(x_i) h  ) \cdot u_{i+1} = 2 h^2 f(x_i))
\end{equation}
